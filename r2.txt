pipeline{
    agent any
    tools{
        maven 'MAVEN_HOME'
    }
    stages{
        stage("git repo & clean"){
            steps{
                bat "git clone https://github.com/bhavagna06/Maven.git"
                bat "mvn clean -f Maven"
            }
        }
        stage("install"){
            steps{
                bat "mvn install -f Maven"
            }
        }
        stage("test"){
            steps{
                bat "mvn test -f Maven"
            }
        }
        stage("packaging"){
            steps{
                bat "mvn package -f Maven"
            }
        }
    }
}
